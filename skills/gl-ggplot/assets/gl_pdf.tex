% machines that haven't kept tlmgr up to date.

% ---- Headings ---------------------------------------------------------------
% Brand-blue section headings, plain dark subsections, muted sub-subsections.
% Avoids titlesec — uses LaTeX's built-in \@startsection redefinition via
% \makeatletter so we keep the dependency surface small.

\makeatletter
\renewcommand\section{%
  \@startsection{section}{1}{\z@}%
    {-2.6ex \@plus -1ex \@minus -.2ex}%
    {1.0ex \@plus.2ex}%
    {\sffamily\Large\bfseries\color{glblue}}}
\renewcommand\subsection{%
  \@startsection{subsection}{2}{\z@}%
    {-2.0ex \@plus -1ex \@minus -.2ex}%
    {0.6ex \@plus.2ex}%
    {\sffamily\large\bfseries\color{gldark}}}
\renewcommand\subsubsection{%
  \@startsection{subsubsection}{3}{\z@}%
    {-1.5ex \@plus -1ex \@minus -.2ex}%
    {0.5ex \@plus.2ex}%
    {\sffamily\normalsize\bfseries\color{glmuted}}}
\makeatother

% ---- Links ------------------------------------------------------------------

\usepackage{hyperref}
\hypersetup{
  colorlinks = true,
  linkcolor  = glblue,
  citecolor  = glblue,
  urlcolor   = glblue
}

% ---- Body text and spacing --------------------------------------------------
% Use plain LaTeX primitives instead of parskip/setspace.

\setlength{\parskip}{0.6em}
\setlength{\parindent}{0pt}
\linespread{1.1}

% ---- Captions ---------------------------------------------------------------

\usepackage{caption}
\captionsetup{
  font       = {small, color=glmuted},
  labelfont  = {bf, color=gldark},
  labelsep   = colon,
  margin     = 4pt
}
